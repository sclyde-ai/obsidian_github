- arguments
    - $0 script name
    - $1, 2, ...　引数 arguments
    - "$#" 引数の数 the number of arguments
- operator
    - integer
        - -eq : equal
        - -ne : not equal
        - -gt : greater than
        - -lt : less than
        - -ge : greater than or equal
        - -le : less than or equal
    - string
        - -n : not empty
        - -z : zero lenth
        - == : equal
        - != : not equal
    - file
        - -e : exist file or dir
        - -f : exist a file
        - -d : exist a dir
        - -s : exist the file that size greater than 0
        - -r : readable
        - -w : writable
        - -x : executable
- grammar
    - [] : classical
    - [[]] : modern 
    - (()) : it is the same to C lang
    - if 
        - if [[condition]]; then
            - ...
        - elif [[condition]]: then
            - ...
        - else
            - ...
        - fi
    - and
        - [[ && ]]
    - or
        - [[ || ]]
    - not
        - [[ ! ]]
